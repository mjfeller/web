     61 static void
     62 appenditem(struct item *item, struct item **list, struct item **last)
     63 {
     64 	if (*last)
     65 		(*last)->right = item;
     66 	else
     67 		*list = item;
     68 
     69 	item->left = *last;
     70 	item->right = NULL;
     71 	*last = item;
     72 }

     74 static void
     75 calcoffsets(void)
     76 {
     77 	int i, n;
     78 
     79 	if (lines > 0)
     80 		n = lines * bh;
     81 	else
     82 		n = mw - (promptw + inputw + TEXTW("<") + TEXTW(">"));
     83 	/* calculate which items will begin the next page and previous page */
     84 	for (i = 0, next = curr; next; next = next->right)
     85 		if ((i += (lines > 0) ? bh : MIN(TEXTW(next->text), n)) > n)
     86 			break;
     87 	for (i = 0, prev = curr; prev && prev->left; prev = prev->left)
     88 		if ((i += (lines > 0) ? bh : MIN(TEXTW(prev->left->text), n)) > n)
     89 			break;
     90 }

    125 static int
    126 drawitem(struct item *item, int x, int y, int w)
    127 {
    128 	if (item == sel)
    129 		drw_setscheme(drw, scheme[SchemeSel]);

    138 static void
    139 drawmenu(void)
    140 {
    141 	unsigned int curpos;
    142 	struct item *item;
    143 	int x = 0, y = 0, fh = drw->fonts->h, w;
    144 
    145 	drw_setscheme(drw, scheme[SchemeNorm]);
    146 	drw_rect(drw, 0, 0, mw, mh, 1, 1);
    147 
    148 	if (prompt && *prompt) {
    149 		drw_setscheme(drw, scheme[SchemeSel]);
    150 		x = drw_text(drw, x, 0, promptw, bh, lrpad / 2, prompt, 0);
    151 	}
    152 	/* draw input field */
    153 	w = (lines > 0 || !matches) ? mw - x : inputw;
    154 	drw_setscheme(drw, scheme[SchemeNorm]);
    155 	drw_text(drw, x, 0, w, bh, lrpad / 2, text, 0);
    156 
    157 	curpos = TEXTW(text) - TEXTW(&text[cursor]);
    158 	if ((curpos += lrpad / 2 - 1) < w) {
    159 		drw_setscheme(drw, scheme[SchemeNorm]);
    160 		drw_rect(drw, x + curpos, 2 + (bh-fh)/2, 2, bh - 4, 1, 0);
    161 	}
    162 
    163 	if (lines > 0) {
    164 		/* draw vertical list */
    165 		for (item = curr; item != next; item = item->right)
    166 			drawitem(item, x, y += bh, mw - x);
    167 	} else if (matches) {
    168 		/* draw horizontal list */
    169 		x += inputw;
    170 		w = TEXTW("<");
    171 		if (curr->left) {
    172 			drw_setscheme(drw, scheme[SchemeNorm]);
    173 			drw_text(drw, x, 0, w, bh, lrpad / 2, "<", 0);
    174 		}
    175 		x += w;
    176 		for (item = curr; item != next; item = item->right)
    177 			x = drawitem(item, x, 0, MIN(TEXTW(item->text), mw - x - TEXTW(">")));
    178 		if (next) {
    179 			w = TEXTW(">");
    180 			drw_setscheme(drw, scheme[SchemeNorm]);
    181 			drw_text(drw, mw - w, 0, w, bh, lrpad / 2, ">", 0);
    182 		}
    183 	}
    184 	drw_map(drw, win, 0, 0, mw, mh);
    185 }

    187 static void
    188 grabfocus(void)
    189 {
    190 	struct timespec ts = { .tv_sec = 0, .tv_nsec = 10000000  };
    191 	Window focuswin;
    192 	int i, revertwin;
    193 
    194 	for (i = 0; i < 100; ++i) {
    195 		XGetInputFocus(dpy, &focuswin, &revertwin);
    196 		if (focuswin == win)
    197 			return;
    198 		XSetInputFocus(dpy, win, RevertToParent, CurrentTime);
    199 		nanosleep(&ts, NULL);
    200 	}
    201 	die("cannot grab focus");
    202 }

    207 	struct timespec ts = { .tv_sec = 0, .tv_nsec = 1000000  };

    222 static void
    223 match(void)
    224 {
    225 	static char **tokv = NULL;
    226 	static int tokn = 0;
    227 
    228 	char buf[sizeof text], *s;
    229 	int i, tokc = 0;
    230 	size_t len, textsize;
    231 	struct item *item, *lprefix, *lsubstr, *prefixend, *substrend;
    232 
    233 	strcpy(buf, text);
    234 	/* separate input text into tokens to be matched individually */
    235 	for (s = strtok(buf, " "); s; tokv[tokc - 1] = s, s = strtok(NULL, " "))
    236 		if (++tokc > tokn && !(tokv = realloc(tokv, ++tokn * sizeof *tokv)))
    237 			die("cannot realloc %u bytes:", tokn * sizeof *tokv);
    238 	len = tokc ? strlen(tokv[0]) : 0;
    239 
    240 	matches = lprefix = lsubstr = matchend = prefixend = substrend = NULL;
    241 	textsize = strlen(text) + 1;
    242 	for (item = items; item && item->text; item++) {
    243 		for (i = 0; i < tokc; i++)
    244 			if (!fstrstr(item->text, tokv[i]))
    245 				break;
    246 		if (i != tokc) /* not all tokens match */
    247 			continue;
    248 		/* exact matches go first, then prefixes, then substrings */
    249 		if (!tokc || !fstrncmp(text, item->text, textsize))
    250 			appenditem(item, &matches, &matchend);
    251 		else if (!fstrncmp(tokv[0], item->text, len))
    252 			appenditem(item, &lprefix, &prefixend);
    253 		else
    254 			appenditem(item, &lsubstr, &substrend);
    255 	}
    256 	if (lprefix) {
    257 		if (matches) {
    258 			matchend->right = lprefix;
    259 			lprefix->left = matchend;
    260 		} else
    261 			matches = lprefix;
    262 		matchend = prefixend;
    263 	}
    264 	if (lsubstr) {
    265 		if (matches) {
    266 			matchend->right = lsubstr;
    267 			lsubstr->left = matchend;
    268 		} else
    269 			matches = lsubstr;
    270 		matchend = substrend;
    271 	}
    272 	curr = sel = matches;
    273 	calcoffsets();
    274 }

    276 static void
    277 insert(const char *str, ssize_t n)
    278 {
    279 	if (strlen(text) + n > sizeof text - 1)
    280 		return;
    281 	/* move existing text out of the way, insert new text, and update cursor */
    282 	memmove(&text[cursor + n], &text[cursor], sizeof text - cursor - MAX(n, 0));
    283 	if (n > 0)
    284 		memcpy(&text[cursor], str, n);
    285 	cursor += n;
    286 	match();
    287 }

    289 static size_t
    290 nextrune(int inc)
    291 {
    292 	ssize_t n;
    293 
    294 	/* return location of next utf8 rune in the given direction (+1 or -1) */
    295 	for (n = cursor + inc; n + inc >= 0 && (text[n] & 0xc0) == 0x80; n += inc)
    296 		;
    297 	return n;
    298 }

    300 static void
    301 movewordedge(int dir)
    302 {
    303 	if (dir < 0) { /* move cursor to the start of the word*/
    304 		while (cursor > 0 && strchr(worddelimiters, text[nextrune(-1)]))
    305 			cursor = nextrune(-1);
    306 		while (cursor > 0 && !strchr(worddelimiters, text[nextrune(-1)]))
    307 			cursor = nextrune(-1);
    308 	} else { /* move cursor to the end of the word */
    309 		while (text[cursor] && strchr(worddelimiters, text[cursor]))
    310 			cursor = nextrune(+1);
    311 		while (text[cursor] && !strchr(worddelimiters, text[cursor]))
    312 			cursor = nextrune(+1);
    313 	}
    314 }

    316 static void
    317 keypress(XKeyEvent *ev)
    318 {
    319 	char buf[32];
    320 	int len;
    321 	KeySym ksym;
    322 	Status status;
    323 
    324 	len = XmbLookupString(xic, ev, buf, sizeof buf, &ksym, &status);
    325 	switch (status) {
    326 	default: /* XLookupNone, XBufferOverflow */
    327 		return;
    328 	case XLookupChars:
    329 		goto insert;
    330 	case XLookupKeySym:
    331 	case XLookupBoth:
    332 		break;
    333 	}
    334 
    335 	if (ev->state & ControlMask) {
    336 		switch(ksym) {
    337 		case XK_a: ksym = XK_Home;      break;
    338 		case XK_b: ksym = XK_Left;      break;
    339 		case XK_c: ksym = XK_Escape;    break;
    340 		case XK_d: ksym = XK_Delete;    break;
    341 		case XK_e: ksym = XK_End;       break;
    342 		case XK_f: ksym = XK_Right;     break;
    343 		case XK_g: ksym = XK_Escape;    break;
    344 		case XK_h: ksym = XK_BackSpace; break;
    345 		case XK_i: ksym = XK_Tab;       break;
    346 		case XK_j: /* fallthrough */
    347 		case XK_J: /* fallthrough */
    348 		case XK_m: /* fallthrough */
    349 		case XK_M: ksym = XK_Return; ev->state &= ~ControlMask; break;
    350 		case XK_n: ksym = XK_Down;      break;
    351 		case XK_p: ksym = XK_Up;        break;
    352 
    353 		case XK_k: /* delete right */
    354 			text[cursor] = '\0';
    355 			match();
    356 			break;
    357 		case XK_u: /* delete left */
    358 			insert(NULL, 0 - cursor);
    359 			break;
    360 		case XK_w: /* delete word */
    361 			while (cursor > 0 && strchr(worddelimiters, text[nextrune(-1)]))
    362 				insert(NULL, nextrune(-1) - cursor);
    363 			while (cursor > 0 && !strchr(worddelimiters, text[nextrune(-1)]))
    364 				insert(NULL, nextrune(-1) - cursor);
    365 			break;
    366 		case XK_y: /* paste selection */
    367 		case XK_Y:
    368 			XConvertSelection(dpy, (ev->state & ShiftMask) ? clip : XA_PRIMARY,
    369 			                  utf8, utf8, win, CurrentTime);
    370 			return;
    371 		case XK_Left:
    372 			movewordedge(-1);
    373 			goto draw;
    374 		case XK_Right:
    375 			movewordedge(+1);
    376 			goto draw;
    377 		case XK_Return:
    378 		case XK_KP_Enter:
    379 			break;
    380 		case XK_bracketleft:
    381 			cleanup();
    382 			exit(1);
    383 		default:
    384 			return;
    385 		}
    386 	} else if (ev->state & Mod1Mask) {
    387 		switch(ksym) {
    388 		case XK_b:
    389 			movewordedge(-1);
    390 			goto draw;
    391 		case XK_f:
    392 			movewordedge(+1);
    393 			goto draw;
    394 		case XK_g: ksym = XK_Home;  break;
    395 		case XK_G: ksym = XK_End;   break;
    396 		case XK_h: ksym = XK_Up;    break;
    397 		case XK_j: ksym = XK_Next;  break;
    398 		case XK_k: ksym = XK_Prior; break;
    399 		case XK_l: ksym = XK_Down;  break;
    400 		default:
    401 			return;
    402 		}
    403 	}
    404 
    405 	switch(ksym) {
    406 	default:
    407 insert:
    408 		if (!iscntrl(*buf))
    409 			insert(buf, len);
    410 		break;
    411 	case XK_Delete:
    412 		if (text[cursor] == '\0')
    413 			return;
    414 		cursor = nextrune(+1);
    415 		/* fallthrough */
    416 	case XK_BackSpace:
    417 		if (cursor == 0)
    418 			return;
    419 		insert(NULL, nextrune(-1) - cursor);
    420 		break;
    421 	case XK_End:
    422 		if (text[cursor] != '\0') {
    423 			cursor = strlen(text);
    424 			break;
    425 		}
    426 		if (next) {
    427 			/* jump to end of list and position items in reverse */
    428 			curr = matchend;
    429 			calcoffsets();
    430 			curr = prev;
    431 			calcoffsets();
    432 			while (next && (curr = curr->right))
    433 				calcoffsets();
    434 		}
    435 		sel = matchend;
    436 		break;
    437 	case XK_Escape:
    438 		cleanup();
    439 		exit(1);
    440 	case XK_Home:
    441 		if (sel == matches) {
    442 			cursor = 0;
    443 			break;
    444 		}
    445 		sel = curr = matches;
    446 		calcoffsets();
    447 		break;
    448 	case XK_Left:
    449 		if (cursor > 0 && (!sel || !sel->left || lines > 0)) {
    450 			cursor = nextrune(-1);
    451 			break;
    452 		}
    453 		if (lines > 0)
    454 			return;
    455 		/* fallthrough */
    456 	case XK_Up:
    457 		if (sel && sel->left && (sel = sel->left)->right == curr) {
    458 			curr = prev;
    459 			calcoffsets();
    460 		}
    461 		break;
    462 	case XK_Next:
    463 		if (!next)
    464 			return;
    465 		sel = curr = next;
    466 		calcoffsets();
    467 		break;
    468 	case XK_Prior:
    469 		if (!prev)
    470 			return;
    471 		sel = curr = prev;
    472 		calcoffsets();
    473 		break;
    474 	case XK_Return:
    475 	case XK_KP_Enter:
    476 		puts((sel && !(ev->state & ShiftMask)) ? sel->text : text);
    477 		if (!(ev->state & ControlMask)) {
    478 			cleanup();
    479 			exit(0);
    480 		}
    481 		if (sel)
    482 			sel->out = 1;
    483 		break;
    484 	case XK_Right:
    485 		if (text[cursor] != '\0') {
    486 			cursor = nextrune(+1);
    487 			break;
    488 		}
    489 		if (lines > 0)
    490 			return;
    491 		/* fallthrough */
    492 	case XK_Down:
    493 		if (sel && sel->right && (sel = sel->right) == next) {
    494 			curr = next;
    495 			calcoffsets();
    496 		}
    497 		break;
    498 	case XK_Tab:
    499 		if (!sel)
    500 			return;
    501 		strncpy(text, sel->text, sizeof text - 1);
    502 		text[sizeof text - 1] = '\0';
    503 		cursor = strlen(text);
    504 		match();
    505 		break;
    506 	}
    507 
    508 draw:
    509 	drawmenu();
    510 }

    512 static void
    513 paste(void)
    514 {
    515 	char *p, *q;
    516 	int di;
    517 	unsigned long dl;
    518 	Atom da;
    519 
    520 	/* we have been given the current selection, now insert it into input */
    521 	if (XGetWindowProperty(dpy, win, utf8, 0, (sizeof text / 4) + 1, False,
    522 	                   utf8, &da, &di, &dl, &dl, (unsigned char **)&p)
    523 	    == Success && p) {
    524 		insert(p, (q = strchr(p, '\n')) ? q - p : (ssize_t)strlen(p));
    525 		XFree(p);
    526 	}
    527 	drawmenu();
    528 }

    530 static void
    531 readstdin(void)
    532 {
    533 	char buf[sizeof text], *p;
    534 	size_t i, imax = 0, size = 0;
    535 	unsigned int tmpmax = 0;
    536 
    537 	/* read each line from stdin and add it to the item list */
    538 	for (i = 0; fgets(buf, sizeof buf, stdin); i++) {
    539 		if (i + 1 >= size / sizeof *items)
    540 			if (!(items = realloc(items, (size += BUFSIZ))))
    541 				die("cannot realloc %u bytes:", size);
    542 		if ((p = strchr(buf, '\n')))
    543 			*p = '\0';
    544 		if (!(items[i].text = strdup(buf)))
    545 			die("cannot strdup %u bytes:", strlen(buf) + 1);
    546 		items[i].out = 0;
    547 		drw_font_getexts(drw->fonts, buf, strlen(buf), &tmpmax, NULL);
    548 		if (tmpmax > inputw) {
    549 			inputw = tmpmax;
    550 			imax = i;
    551 		}
    552 	}
    553 	if (items)
    554 		items[i].text = NULL;
    555 	inputw = items ? TEXTW(items[imax].text) : 0;
    556 	lines = MIN(lines, i);
    557 }

    559 static void
    560 run(void)
    561 {
    562 	XEvent ev;
    563 
    564 	while (!XNextEvent(dpy, &ev)) {
    565 		if (XFilterEvent(&ev, win))
    566 			continue;
    567 		switch(ev.type) {
    568 		case DestroyNotify:
    569 			if (ev.xdestroywindow.window != win)
    570 				break;
    571 			cleanup();
    572 			exit(1);
    573 		case Expose:
    574 			if (ev.xexpose.count == 0)
    575 				drw_map(drw, win, 0, 0, mw, mh);
    576 			break;
    577 		case FocusIn:
    578 			/* regrab focus from parent window */
    579 			if (ev.xfocus.window != win)
    580 				grabfocus();
    581 			break;
    582 		case KeyPress:
    583 			keypress(&ev.xkey);
    584 			break;
    585 		case SelectionNotify:
    586 			if (ev.xselection.property == utf8)
    587 				paste();
    588 			break;
    589 		case VisibilityNotify:
    590 			if (ev.xvisibility.state != VisibilityUnobscured)
    591 				XRaiseWindow(dpy, win);
    592 			break;
    593 		}
    594 	}
    595 }

    597 static void
    598 setup(void)
    599 {
    600 	int x, y, i, j;
    601 	unsigned int du;
    602 	XSetWindowAttributes swa;
    603 	XIM xim;
    604 	Window w, dw, *dws;
    605 	XWindowAttributes wa;
    606 	XClassHint ch = {"dmenu", "dmenu"};
    607 #ifdef XINERAMA
    608 	XineramaScreenInfo *info;
    609 	Window pw;
    610 	int a, di, n, area = 0;
    611 #endif
    612 	/* init appearance */
    613 	for (j = 0; j < SchemeLast; j++)
    614 		scheme[j] = drw_scm_create(drw, colors[j], 2);
    615 
    616 	clip = XInternAtom(dpy, "CLIPBOARD",   False);
    617 	utf8 = XInternAtom(dpy, "UTF8_STRING", False);
    618 
    619 	/* calculate menu geometry */
    620 	bh = drw->fonts->h + 2;
    621 	bh = MAX(bh,lineheight);	/* make a menu line AT LEAST 'lineheight' tall */
    622 	lines = MAX(lines, 0);
    623 	mh = (lines + 1) * bh;
    624 	promptw = (prompt && *prompt) ? TEXTW(prompt) - lrpad / 4 : 0;
    625 #ifdef XINERAMA
    626 	i = 0;
    627 	if (parentwin == root && (info = XineramaQueryScreens(dpy, &n))) {
    628 		XGetInputFocus(dpy, &w, &di);
    629 		if (mon >= 0 && mon < n)
    630 			i = mon;
    631 		else if (w != root && w != PointerRoot && w != None) {
    632 			/* find top-level window containing current input focus */
    633 			do {
    634 				if (XQueryTree(dpy, (pw = w), &dw, &w, &dws, &du) && dws)
    635 					XFree(dws);
    636 			} while (w != root && w != pw);
    637 			/* find xinerama screen with which the window intersects most */
    638 			if (XGetWindowAttributes(dpy, pw, &wa))
    639 				for (j = 0; j < n; j++)
    640 					if ((a = INTERSECT(wa.x, wa.y, wa.width, wa.height, info[j])) > area) {
    641 						area = a;
    642 						i = j;
    643 					}
    644 		}
    645 		/* no focused window is on screen, so use pointer location instead */
    646 		if (mon < 0 && !area && XQueryPointer(dpy, root, &dw, &dw, &x, &y, &di, &di, &du))
    647 			for (i = 0; i < n; i++)
    648 				if (INTERSECT(x, y, 1, 1, info[i]))
    649 					break;
    650 
    651 		if (centered) {
    652 			mw = MIN(MAX(max_textw() + promptw, min_width), info[i].width);
    653 			x = info[i].x_org + ((info[i].width  - mw) / 2);
    654 			y = info[i].y_org + ((info[i].height - mh) / 2);
    655 		} else {
    656 			x = info[i].x_org;
    657 			y = info[i].y_org + (topbar ? 0 : info[i].height - mh);
    658 			mw = info[i].width;
    659 		}
    660 
    661 		XFree(info);
    662 	} else
    663 #endif
    664 	{
    665 		if (!XGetWindowAttributes(dpy, parentwin, &wa))
    666 			die("could not get embedding window attributes: 0x%lx",
    667 			    parentwin);
    668 
    669 		if (centered) {
    670 			mw = MIN(MAX(max_textw() + promptw, min_width), wa.width);
    671 			x = (wa.width  - mw) / 2;
    672 			y = (wa.height - mh) / 2;
    673 		} else {
    674 			x = 0;
    675 			y = topbar ? 0 : wa.height - mh;
    676 			mw = wa.width;
    677 		}
    678 	}
    679 	inputw = MIN(inputw, mw/3);
    680 	match();
    681 
    682 	/* create menu window */
    683 	swa.override_redirect = True;
    684 	swa.background_pixel = scheme[SchemeNorm][ColBg].pixel;
    685 	swa.event_mask = ExposureMask | KeyPressMask | VisibilityChangeMask;
    686 	win = XCreateWindow(dpy, parentwin, x, y, mw, mh, border_width,
    687 	                    CopyFromParent, CopyFromParent, CopyFromParent,
    688 	                    CWOverrideRedirect | CWBackPixel | CWEventMask, &swa);
    689 	if (border_width)
    690 		XSetWindowBorder(dpy, win, scheme[SchemeSel][ColBg].pixel);
    691 	XSetClassHint(dpy, win, &ch);
    692 
    693 
    694 	/* input methods */
    695 	if ((xim = XOpenIM(dpy, NULL, NULL, NULL)) == NULL)
    696 		die("XOpenIM failed: could not open input device");
    697 
    698 	xic = XCreateIC(xim, XNInputStyle, XIMPreeditNothing | XIMStatusNothing,
    699 	                XNClientWindow, win, XNFocusWindow, win, NULL);
    700 
    701 	XMapRaised(dpy, win);
    702 	if (embed) {
    703 		XSelectInput(dpy, parentwin, FocusChangeMask | SubstructureNotifyMask);
    704 		if (XQueryTree(dpy, parentwin, &dw, &w, &dws, &du) && dws) {
    705 			for (i = 0; i < du && dws[i] != win; ++i)
    706 				XSelectInput(dpy, dws[i], FocusChangeMask);
    707 			XFree(dws);
    708 		}
    709 		grabfocus();
    710 	}
    711 	drw_resize(drw, mw, mh);
    712 	drawmenu();
    713 }

    715 static void
    716 usage(void)
    717 {
    718 	fputs("usage: dmenu [-bfiv] [-l lines] [-p prompt] [-fn font] [-m monitor]\n"
    719 	      "             [-h height]\n"
    720 	      "             [-nb color] [-nf color] [-sb color] [-sf color] [-w windowid]\n", stderr);
    721 	exit(1);
    722 }
    723 
    724 int
    725 main(int argc, char *argv[])
    726 {
    727 	XWindowAttributes wa;
    728 	int i, fast = 0;
    729 
    730 	for (i = 1; i < argc; i++)
    731 		/* these options take no arguments */
    732 		if (!strcmp(argv[i], "-v")) {      /* prints version information */
    733 			puts("dmenu-"VERSION);
    734 			exit(0);
    735 		} else if (!strcmp(argv[i], "-b")) /* appears at the bottom of the screen */
    736 			topbar = 0;
    737 		else if (!strcmp(argv[i], "-f"))   /* grabs keyboard before reading stdin */
    738 			fast = 1;
    739 		else if (!strcmp(argv[i], "-c"))   /* centers dmenu on screen */
    740 			centered = 1;
    741 		else if (!strcmp(argv[i], "-i")) { /* case-insensitive item matching */
    742 			fstrncmp = strncasecmp;
    743 			fstrstr = cistrstr;
    744 		} else if (i + 1 == argc)
    745 			usage();
    746 		/* these options take one argument */
    747 		else if (!strcmp(argv[i], "-l"))   /* number of lines in vertical list */
    748 			lines = atoi(argv[++i]);
    749 		else if (!strcmp(argv[i], "-m"))
    750 			mon = atoi(argv[++i]);
    751 		else if (!strcmp(argv[i], "-p"))   /* adds prompt to left of input field */
    752 			prompt = argv[++i];
    753 		else if (!strcmp(argv[i], "-fn"))  /* font or font set */
    754 			fonts[0] = argv[++i];
    755 		else if(!strcmp(argv[i], "-h")) { /* minimum height of one menu line */
    756 			lineheight = atoi(argv[++i]);
    757 			lineheight = MAX(lineheight,8); /* reasonable default in case of value too small/negative */
    758 		}
    759 		else if (!strcmp(argv[i], "-nb"))  /* normal background color */
    760 			colors[SchemeNorm][ColBg] = argv[++i];
    761 		else if (!strcmp(argv[i], "-nf"))  /* normal foreground color */
    762 			colors[SchemeNorm][ColFg] = argv[++i];
    763 		else if (!strcmp(argv[i], "-sb"))  /* selected background color */
    764 			colors[SchemeSel][ColBg] = argv[++i];
    765 		else if (!strcmp(argv[i], "-sf"))  /* selected foreground color */
    766 			colors[SchemeSel][ColFg] = argv[++i];
    767 		else if (!strcmp(argv[i], "-w"))   /* embedding window id */
    768 			embed = argv[++i];
    769 		else if (!strcmp(argv[i], "-bw"))
    770 			border_width = atoi(argv[++i]); /* border width */
    771 		else
    772 			usage();
    773 
    774 	if (!setlocale(LC_CTYPE, "") || !XSupportsLocale())
    775 		fputs("warning: no locale support\n", stderr);
    776 	if (!(dpy = XOpenDisplay(NULL)))
    777 		die("cannot open display");
    778 	screen = DefaultScreen(dpy);
    779 	root = RootWindow(dpy, screen);
    780 	if (!embed || !(parentwin = strtol(embed, NULL, 0)))
    781 		parentwin = root;
    782 	if (!XGetWindowAttributes(dpy, parentwin, &wa))
    783 		die("could not get embedding window attributes: 0x%lx",
    784 		    parentwin);
    785 	drw = drw_create(dpy, screen, root, wa.width, wa.height);
    786 	if (!drw_fontset_create(drw, fonts, LENGTH(fonts)))
    787 		die("no fonts could be loaded.");
    788 	lrpad = drw->fonts->h;
    789 
    790 #ifdef __OpenBSD__
    791 	if (pledge("stdio rpath", NULL) == -1)
    792 		die("pledge");
    793 #endif
    794 
    795 	if (fast && !isatty(0)) {
    796 		grabkeyboard();
    797 		readstdin();
    798 	} else {
    799 		readstdin();
    800 		grabkeyboard();
    801 	}
    802 	setup();
    803 	run();
    804 
    805 	return 1; /* unreachable */
    806 }
