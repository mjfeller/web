     79          (defun ibuffer-get-major-modes-ibuff-rules-list (mm-list result-list)
     80            (if mm-list
     81                (let* ((cur-mm (car mm-list))
     82                       (next-res-list-el `(,(symbol-name cur-mm) (mode . ,cur-mm))))
     83                  (ibuffer-get-major-modes-ibuff-rules-list
     84                   (cdr mm-list) (cons next-res-list-el result-list)))
     85              result-list))
     86 
     87          (defun ibuffer-get-major-modes-list ()
     88            (mapcar
     89             (function (lambda (buffer)
     90                         (buffer-local-value 'major-mode (get-buffer buffer))))
     91             (buffer-list (selected-frame))))
     92 
     93          (defun ibuffer-create-buffs-group ()
     94            (interactive)
     95            (let* ((ignore-modes '(Buffer-menu-mode
     96                                   compilation-mode
     97                                   minibuffer-inactive-mode
     98                                   ibuffer-mode
     99                                   magit-process-mode
    100                                   messages-buffer-mode
    101                                   fundamental-mode
    102                                   completion-list-mode
    103                                   help-mode
    104                                   Info-mode))
    105                   (cur-bufs
    106                    (list (cons "Home"
    107                                (ibuffer-get-major-modes-ibuff-rules-list
    108                                 (cl-set-difference
    109                                  (remove-duplicates
    110                                   (ibuffer-get-major-modes-list))
    111                                  ignore-modes) '())))))
    112              (setq ibuffer-saved-filter-groups cur-bufs)
    113              (ibuffer-switch-to-saved-filter-groups "Home")))

    145          ;; Use human readable Size column instead of original one
    146          (define-ibuffer-column size-h
    147            (:name "Size" :inline t)
    148            (cond
    149             ((> (buffer-size) 1000000) (format "%7.1fM" (/ (buffer-size) 1000000.0)))
    150             ((> (buffer-size) 100000) (format "%7.0fk" (/ (buffer-size) 1000.0)))
    151             ((> (buffer-size) 1000) (format "%7.1fk" (/ (buffer-size) 1000.0)))
    152             (t (format "%8d" (buffer-size)))))
