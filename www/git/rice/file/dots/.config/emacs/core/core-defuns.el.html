     37 (defun transpose-words (arg)
     38   "[Override for default transpose-words in simple.el]
     39               Interchange words around point, leaving point at end of
     40               them. With prefix arg ARG, effect is to take word before or
     41               around point and drag it backward past ARG other words (forward
     42               if ARG negative).  If ARG is zero, the words around or after
     43               point and around or after mark are interchanged."
     44   (interactive "*p")
     45   (if (eolp) (forward-char -1))
     46   (transpose-subr 'backward-word arg)
     47   (forward-word (+ arg 1)))

     65 (defun toggle-window-split ()
     66   "Toggle the current window split."
     67   (interactive)
     68   (if (= (count-windows) 2)
     69       (let* ((this-win-buffer (window-buffer))
     70              (next-win-buffer (window-buffer (next-window)))
     71              (this-win-edges (window-edges (selected-window)))
     72              (next-win-edges (window-edges (next-window)))
     73              (this-win-2nd (not (and (<= (car this-win-edges)
     74                                          (car next-win-edges))
     75                                      (<= (cadr this-win-edges)
     76                                          (cadr next-win-edges)))))
     77              (splitter
     78               (if (= (car this-win-edges)
     79                      (car (window-edges (next-window))))
     80                   'split-window-horizontally
     81                 'split-window-vertically)))
     82         (delete-other-windows)
     83         (let ((first-win (selected-window)))
     84           (funcall splitter)
     85           (if this-win-2nd (other-window 1))
     86           (set-window-buffer (selected-window) this-win-buffer)
     87           (set-window-buffer (next-window) next-win-buffer)
     88           (select-window first-win)
     89           (if this-win-2nd (other-window 1))))))

    105 (defadvice move-beginning-of-line (around smarter-bol activate)
    106   ;; Move to requested line if needed.
    107   (let ((arg (or (ad-get-arg 0) 1)))
    108     (when (/= arg 1)
    109       (forward-line (1- arg))))
    110   ;; Move to indentation on first call, then to actual BOL on second.
    111   (let ((pos (point)))
    112     (back-to-indentation)
    113     (when (= pos (point))
    114       ad-do-it)))

    116 ;; If the *scratch* buffer is killed, recreate it automatically
    117 ;; FROM: Morten Welind
    118 ;;http://www.geocrawler.com/archives/3/338/1994/6/0/1877802/
    119 (save-excursion
    120   (set-buffer (get-buffer-create "*scratch*"))
    121   (lisp-interaction-mode)
    122   (make-local-variable 'kill-buffer-query-functions)
    123   (add-hook 'kill-buffer-query-functions 'kill-scratch-buffer))

    125 (defun kill-scratch-buffer ()
    126   ;; The next line is just in case someone calls this manually
    127   (set-buffer (get-buffer-create "*scratch*"))
    128   ;; Kill the current (*scratch*) buffer
    129   (remove-hook 'kill-buffer-query-functions 'kill-scratch-buffer)
    130   (kill-buffer (current-buffer))
    131   ;; Make a brand new *scratch* buffer
    132   (set-buffer (get-buffer-create "*scratch*"))
    133   (lisp-interaction-mode)
    134   (make-local-variable 'kill-buffer-query-functions)
    135   (add-hook 'kill-buffer-query-functions 'kill-scratch-buffer)
    136   ;; Since we killed it, don't let caller do that.
    137   nil)

    139 (defun save-buffer-without-dtw ()
    140   (interactive)
    141   (let ((b (current-buffer)))   ; memorize the buffer
    142     (with-temp-buffer ; new temp buffer to bind the global value of before-save-hook
    143       (let ((before-save-hook (remove 'delete-trailing-whitespace before-save-hook)))
    144         (with-current-buffer b  ; go back to the current buffer, before-save-hook is now buffer-local
    145           (let ((before-save-hook (remove 'delete-trailing-whitespace before-save-hook)))
    146             (save-buffer)))))))

    148 (defun narrow-or-widen-dwim (p)
    149   "Widen if buffer is narrowed, narrow-dwim otherwise.
    150 Dwim means: region, org-src-block, org-subtree, or
    151 defun, whichever applies first. Narrowing to
    152 org-src-block actually calls `org-edit-src-code'.
    153 
    154 With prefix P, don't widen, just narrow even if buffer
    155 is already narrowed."
    156   (interactive "P")
    157   (declare (interactive-only))
    158   (cond ((and (buffer-narrowed-p) (not p)) (widen))
    159         ((region-active-p)
    160          (narrow-to-region (region-beginning)
    161                            (region-end)))
    162         ((derived-mode-p 'org-mode)
    163          ;; `org-edit-src-code' is not a real narrowing
    164          ;; command. Remove this first conditional if
    165          ;; you don't want it.
    166          (cond ((ignore-errors (org-edit-src-code) t)
    167                 (delete-other-windows))
    168                ((ignore-errors (org-narrow-to-block) t))
    169                (t (org-narrow-to-subtree))))
    170         ((derived-mode-p 'latex-mode)
    171          (LaTeX-narrow-to-environment))
    172         (t (narrow-to-defun))))
    173 
    174 ;; (define-key endless/toggle-map "n"
    175 ;;   #'narrow-or-widen-dwim)
    176 ;; This line actually replaces Emacs' entire narrowing
    177 ;; keymap, that's how much I like this command. Only
    178 ;; copy it if that's what you want.
    179 (define-key ctl-x-map "n" #'narrow-or-widen-dwim)
    180 (add-hook 'LaTeX-mode-hook
    181           (lambda ()
    182             (define-key LaTeX-mode-map "\C-xn"
    183               nil)))
