     43 (defun finder ()
     44   "Opens file directory in Finder."
     45   (interactive)
     46   (let ((file (buffer-file-name)))
     47     (if file
     48         (shell-command
     49          (format "%s %s" (executable-find "open") (file-name-directory file)))
     50       (error "Buffer is not attached to any file."))))

     52 (when (not (display-graphic-p))
     53   (defun bw/copy-from-osx ()
     54     "Copies the current clipboard content using the `pbcopy` command"
     55     (shell-command-to-string "pbpaste"))
     56 
     57   (defun bw/paste-to-osx (text &optional push)
     58     "Copies the top of the kill ring stack to the OSX clipboard"
     59     (let ((process-connection-type nil))
     60       (let ((proc (start-process "pbcopy" "*Messages*" "pbcopy")))
     61         (process-send-string proc text)
     62         (process-send-eof proc))))
     63 
     64   (setq interprogram-cut-function 'bw/paste-to-osx)
     65   (setq interprogram-paste-function 'bw/copy-from-osx))
