     24 ;;---------------------------------------------------------------------
     25 ;; Init

     28 ;; Added by Package.el.  This must come before configurations of
     29 ;; installed packages.  Don't delete this line.  If you don't want it,
     30 ;; just comment it out by adding a semicolon to the start of the line.
     31 ;; You may delete these explanatory comments.
     32 (package-initialize)
     33 
     34 (defvar current-user
     35   (getenv
     36    (if (equal system-type 'windows-nt) "USERNAME" "USER")))
     37 
     38 (message "Powering up... Be patient, Master %s!" current-user)
     39 
     40 (when (version< emacs-version "24.4")
     41   (error "Requires at least GNU Emacs 24.4, but you're running %s" emacs-version))
     42 
     43 ;; Always load newest byte code
     44 (setq load-prefer-newer t)
     45 
     46 ;; reduce the frequency of garbage collection by making it happen on
     47 ;; each 50MB of allocated data (the default is on every 0.76MB)
     48 (setq gc-cons-threshold 50000000)
     49 
     50 ;; warn when opening files bigger than 100MB
     51 (setq large-file-warning-threshold 100000000)
     52 
     53 (setq inhibit-startup-echo-area-message t)
     54 
     55 ;;---------------------------------------------------------------------
     56 ;; Variables and Load Paths
     57 ;;---------------------------------------------------------------------
     58 
     59 (defvar emacs-dir (file-name-directory "~/.emacs.d/")
     60   "The root dir of the Emacs distribution.")
     61 
     62 (defvar core-dir (expand-file-name "core" emacs-dir)
     63   "The home of core functionality.")
     64 
     65 (defvar modules-dir (expand-file-name "modules" emacs-dir)
     66   "This directory houses all of the modules.")
     67 
     68 (defvar persistent-dir (expand-file-name "etc" emacs-dir)
     69   "This directory houses files created by emacs packages")
     70 
     71 (defvar lisp-dir (expand-file-name "lisp" emacs-dir)
     72   "This directory houses user lisp and site lisp")
     73 
     74 (unless (file-exists-p persistent-dir)
     75   (make-directory persistent-dir))
     76 
     77 (if (file-exists-p (concat emacs-dir "src"))
     78   (setq source-directory (concat emacs-dir "src")))
     79 
     80 (add-to-list 'load-path core-dir)
     81 (add-to-list 'load-path modules-dir)
     82 (add-to-list 'load-path lisp-dir)
     83 
     84 ;; load the custom file
     85 (setq custom-file (concat persistent-dir "/custom.el"))
     86 (unless (file-exists-p custom-file)
     87   (write-region "" nil custom-file))
     88 (load custom-file)
     89 
     90 (require 'init-preload-local nil t)
     91 
     92 ;;---------------------------------------------------------------------
     93 ;; Core
     94 ;;---------------------------------------------------------------------
     95 
     96 (message "Loading core...")
     97 
     98 (require 'core-bootstrap)
     99 (require 'core-ui)
    100 (when (eq system-type 'darwin) (require 'core-osx))
    101 (require 'core-packages)
    102 (require 'core-defuns)
    103 (require 'core-paths)
    104 
    105 ;;---------------------------------------------------------------------
    106 ;; Modules

    111 ;; (require 'module-bbdb)
    112 (require 'module-c)
    113 (require 'module-company)
    114 (require 'module-docker)
    115 (require 'module-eldoc)
    116 (require 'module-emacs-lisp)
    117 (require 'module-evil)
    118 (require 'module-git)
    119 (require 'module-go)
    120 (require 'module-multiple-cursors)
    121 (require 'module-org)
    122 (require 'module-projectile)
    123 (require 'module-protobuf)
    124 (require 'module-python)
    125 (require 'module-restclient)
    126 (require 'module-yaml)
    127 (require 'module-yasnippet)
    128 
    129 ;; load any custom user provided locals
    130 (require 'init-local nil t)
    131 
    132 (message "Ready to do thy bidding, Master %s!" current-user)
