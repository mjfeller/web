     31 ;; Emacs extension for controlling Kubernetes with limited permissions.
     32 ;; Mostly focuses on pod management for now.
     33 
     34 ;;; Usage:
     35 
     36 ;; To list the pods in your current context and namespace, call
     37 ;;
     38 ;; M-x kubel
     39 ;;
     40 ;; To set said namespace and context, respectively call
     41 ;;
     42 ;; M-x kubel-set-namespace
     43 ;; M-x kubel-set-context
     44 ;;
     45 ;; Note that namespace will autocomplete but not context,
     46 ;; this is because I interact with kubernetes through a user who
     47 ;; does not have permissions to list namespaces.
     48 
     49 ;;; Shortcuts:
     50 
     51 ;; On the kubel screen, place your cursor on the pod
     52 ;;
     53 ;; enter => get pod details
     54 ;; h => help popup
     55 ;; C => set context
     56 ;; n => set namespace
     57 ;; g => refresh pods
     58 ;; p => port forward pod
     59 ;; e => exec into pod
     60 ;; d => describe popup
     61 ;; l => log popup
     62 ;; c => copy popup
     63 ;; k => delete pod
     64 ;; j => jab deployment to force rolling update

    124 (defun kubel--list-entries ()
    125   "Create the entries for the service list."
    126   (let ((temp (list)))
    127     (with-temp-buffer
    128       (insert (shell-command-to-string (concat (kubel--get-command-prefix) " get pods --no-headers=true")))
    129       (goto-char (point-min))
    130       (while (re-search-forward "^\\([a-z0-9\-]+\\) +\\([0-9]+/[0-9]+\\) +\\(\\w+\\) +\\([0-9]+\\) +\\([0-9a-z]+\\)$" (point-max) t)
    131         (setq temp (append temp (list (list (match-string 1) (vector (match-string 1) (match-string 2) (kubel--propertize-status (match-string 3)) (match-string 4) (match-string 5)))))))
    132       )
    133     temp))

    136   "Utility function to pop to buffer or create it.

    143 (defun kubel--exec (buffer-name async args)
    144   "Utility function to run commands in the proper context and namespace.
    145 
    146 \"BUFFER-NAME\" is the buffer-name. Default to *kubel-command*.
    147 ASYNC is a bool. If true will run async.
    148 ARGS is a ist of arguments."
    149   (when (equal buffer-name "")
    150     (setq buffer-name "*kubel-command*"))
    151   (when (get-buffer buffer-name)
    152     (kill-buffer buffer-name))
    153   (if async
    154       (apply #'start-process buffer-name buffer-name "kubectl" (append (kubel--get-context-namespace) args))
    155     (apply #'call-process "kubectl" nil buffer-name nil (append (kubel--get-context-namespace) args)))
    156   (pop-to-buffer buffer-name))

    182 (defun kubel--select-resource (name)
    183   "Prompt user to select an instance out of a list of resources.
    184 
    185 NAME is the string name of the resource."
    186   (let ((cmd (format "%s get %s -o=jsonpath='{.items[*].metadata.name}'"
    187                      (kubel--get-command-prefix) name)))
    188     (completing-read (concat (s-upper-camel-case name) ": ")
    189                      (split-string (shell-command-to-string cmd) " "))))

    191 (defun kubel--describe-resource (name &optional yaml)
    192   "Describe a specific resource.
    193 
    194 NAME is the string name of the resource to decribe.
    195 YAML is boolean to show resource as yaml"
    196   (let* ((resource (kubel--select-resource name))
    197          (buffer-name (format "*kubel - %s - %s*" name resource)))
    198     (if yaml
    199         (kubel--exec buffer-name nil (list "get" name "-o" "yaml" resource))
    200       (kubel--exec buffer-name nil (list "describe" name resource)))
    201     (when yaml
    202       (yaml-mode))
    203     (beginning-of-buffer)))

    214 (defun kubel--default-tail-arg (args)
    215   "Ugly function to make sure that there is at least the default tail.
    216 
    217 ARGS is the arg list from transient."
    218   (if (car (remove nil (mapcar (lambda (x)
    219                                  (string-prefix-p "--tail=" x)) args)))
    220       args
    221     (append args (list (concat "--tail=" kubel-log-tail-n)))))

    223 (defun kubel-get-pod-logs (&optional args)
    224   "Get the last N logs of the pod under the cursor.
    225 
    226 ARGS is the arguments list from transient."
    227   (interactive
    228    (list (transient-args 'kubel-log-popup)))
    229   (let* ((pod (kubel--get-pod-under-cursor))
    230          (containers (kubel--get-containers pod))
    231          (container (if (equal (length containers) 1)
    232                         (car containers)
    233                       (completing-read "Select container: " containers)))
    234          (buffer-name (format "*kubel - logs - %s - %s*" pod container))
    235          (async nil))
    236     (when (member "-f" args)
    237       (setq async t))
    238     (kubel--exec buffer-name async
    239                  (append '("logs") (kubel--default-tail-arg args) (list pod container)))))

    347 (defun kubel-exec-pod ()
    348   "Kubectl exec into the pod under the cursor."
    349   (interactive)
    350   (let* ((pod (kubel--get-pod-under-cursor))
    351          (containers (kubel--get-containers pod))
    352          (container (if (equal (length containers) 1)
    353                         (car containers)
    354                       (completing-read "Select container: " containers))))
    355     (eshell)
    356     (insert (format "%s exec -it %s -c %s /bin/sh" (kubel--get-command-prefix) pod container))))

    368 (defun kubel-jab-deployment ()
    369   "Make a trivial patch to force a new deployment.
    370 
    371 See https://github.com/kubernetes/kubernetes/issues/27081"
    372   (interactive)
    373   (let* ((deployment (kubel--select-resource "deployment"))
    374          (buffer-name (format "*kubel - bouncing - %s*" deployment)))
    375     (kubel--exec buffer-name nil (list "patch" "deployment" deployment "-p"
    376                         (format "{\"spec\":{\"template\":{\"metadata\":{\"labels\":{\"date\":\"%s\"}}}}}"
    377                                 (round (time-to-seconds)))))))

    463 (defun kubel ()
    464   "Invoke the kubel buffer."
    465   (interactive)
    466   (kubel--pop-to-buffer (kubel--buffer-name))
    467   (kubel-mode)
    468   (message (concat "Namespace: " kubel-namespace)))

    470 (define-derived-mode kubel-mode tabulated-list-mode "Kubel"
    471   "Special mode for kubel buffers."
    472   (buffer-disable-undo)
    473   (kill-all-local-variables)
    474   (setq truncate-lines t)
    475   (setq mode-name "Kubel")
    476   (setq major-mode 'kubel-mode)
    477   (use-local-map kubel-mode-map)
    478   (setq tabulated-list-format kubel--list-format)
    479   (setq tabulated-list-entries 'kubel--list-entries)
    480   (setq tabulated-list-sort-key kubel--list-sort-key)
    481   (tabulated-list-init-header)
    482   (tabulated-list-print)
    483   (hl-line-mode 1)
    484   (run-mode-hooks 'kubel-mode-hook))
