     94 (defun helm--list-entries ()
     95   "Create the entries for the service list."
     96   (let ((output (helm--exec-string '("ls --output json"))))
     97     (mapcar (lambda (x)
     98               (list (cdr (car x))
     99                     (vconcat (mapcar (lambda (y)
    100                                        (format "%s" (cdr y))) x))))
    101             (cdr (car (cdr (json-read-from-string output)))))))

    103 (defun helm--exec (buffer-name async args)
    104   "Utility function to run commands in the proper context and namespace.
    105 
    106 \"BUFFER-NAME\" is the buffer-name. Default to *helm-command*.
    107 ASYNC is a bool. If true will run async.
    108 ARGS is a ist of arguments."
    109   (when (equal buffer-name "")
    110     (setq buffer-name "*helm-command*"))
    111   (when (get-buffer buffer-name)
    112     (kill-buffer buffer-name))
    113   (if async
    114       (apply #'start-process buffer-name buffer-name "helm" (append (helm--get-context) args))
    115     (apply #'call-process "helm" nil buffer-name nil (append (helm--get-context) args)))
    116   (pop-to-buffer buffer-name))

    123 (defun helm--pop-to-buffer (name)
    124   "Utility function to pop to buffer or create it.
    125 
    126 NAME is the buffer name."
    127   (unless (get-buffer name)
    128     (get-buffer-create name))
    129   (pop-to-buffer name))

    349 (defun helm ()
    350   "Invoke the helm buffer."
    351   (interactive)
    352   (helm--pop-to-buffer (helm--buffer-name))
    353   (helm-mode))

    355 (define-derived-mode helm-mode tabulated-list-mode "Helm"
    356   "Special mode for helm buffers."
    357   (buffer-disable-undo)
    358   (kill-all-local-variables)
    359   (setq truncate-lines t)
    360   (setq mode-name "Helm")
    361   (setq major-mode 'helm-mode)
    362   (use-local-map helm-mode-map)
    363   (setq tabulated-list-format helm--list-format)
    364   (setq tabulated-list-entries 'helm--list-entries)
    365   (setq tabulated-list-sort-key helm--list-sort-key)
    366   (tabulated-list-init-header)
    367   (tabulated-list-print)
    368   (hl-line-mode 1)
    369   (run-mode-hooks 'helm-mode-hook))
