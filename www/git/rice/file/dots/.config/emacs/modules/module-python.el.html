     53 (defun pyenv-init()
     54   (setq global-pyenv (replace-regexp-in-string "\n" "" (shell-command-to-string "pyenv global")))
     55   (message (concat "Setting pyenv version to " global-pyenv))
     56   (pyenv-mode-set global-pyenv)
     57   (defvar pyenv-current-version nil global-pyenv))
     58 
     59 (defun pyenv-activate-current-project ()
     60   "Automatically activates pyenv version if .python-version file exists."
     61   (interactive)
     62   (f-traverse-upwards
     63    (lambda (path)
     64      (message path)
     65      (let ((pyenv-version-path (f-expand ".python-version" path)))
     66        (if (f-exists? pyenv-version-path)
     67            (progn
     68              (setq pyenv-current-version (s-trim (f-read-text pyenv-version-path 'utf-8)))
     69              (pyenv-mode-set pyenv-current-version)
     70              (pyvenv-workon pyenv-current-version)
     71              (message (concat "Setting virtualenv to " pyenv-current-version))))))))
