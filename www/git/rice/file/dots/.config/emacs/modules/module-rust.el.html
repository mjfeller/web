     24 (use-package rust-mode
     25   :bind (:map rust-mode-map
     26               ("C-c C-c"   . compile)
     27               ("C-c <tab>" . rust-format-buffer))
     28   :config
     29   (progn (setq rust-format-on-save t)
     30 
     31          (defun rust-prog-init ()
     32              "if cargo is install get all depencies for emacs rust packages"
     33              (interactive)
     34              (if (not (eq (executable-find "cargo") nil))
     35                  (progn
     36                    (mapcar (lambda (pkg)
     37                              (let ((cmd (concat "cargo install " pkg)))
     38                                (call-process-shell-command cmd nil "*cargo-output*" t)))
     39                            '("racer")))
     40                (message "cargo executable not found")))
     41 
     42          (defun setup-rust-mode-compile ()
     43            "Customize compile command to run go build"
     44            (if (not (string-match "cargo" compile-command))
     45                (set (make-local-variable 'compile-command)
     46                     "cargo build")))
     47 
     48          (add-hook 'rust-mode-hook 'setup-rust-mode-compile)))
