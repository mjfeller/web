     99 (define-minor-mode minor-mode-blackout-mode
    100   "Hides minor modes from the mode line."
    101   t)
    102 
    103 (catch 'done
    104   (mapc (lambda (x)
    105           (when (and (consp x)
    106                      (equal (cadr x) '("" minor-mode-alist)))
    107             (let ((original (copy-sequence x)))
    108               (setcar x 'minor-mode-blackout-mode)
    109               (setcdr x (list "" original)))
    110             (throw 'done t)))
    111         mode-line-modes))
