     31 (use-package go-mode
     32   :bind (:map go-mode-map
     33               ("C-c C-g" . go-goto-imports)
     34               ("C-c C-c" . compile)

     42          (defun go-prog-init ()
     43            "if go is install go get all dependencies for emacs go packages"
     44            (interactive)
     45            (if (not (eq (executable-find "go") nil))
     46                (progn
     47                  (mapcar (lambda (pkg)
     48                            (let ((cmd (concat "go get -u " pkg)))
     49                              (call-process-shell-command cmd nil "*go-get-output*" t)))
     50                          '("github.com/nsf/gocode"
     51                            "golang.org/x/tools/cmd/goimports"
     52                            "github.com/rogpeppe/godef"
     53                            "github.com/golang/lint"
     54                            "golang.org/x/tools/cmd/gorename"
     55                            "golang.org/x/tools/cmd/guru"
     56                            "github.com/kisielk/errcheck")))
     57              (message "go executable not found, install go from https://golang.org/download")))
     58 
     59          (defun ginkgo-test (package)
     60            "Run ginkgo test suites recursively from a given go package"
     61            (interactive (list (read-directory-name "Package:")))
     62            (async-shell-command
     63             (concat "ginkgo -r " package) "*Ginkgo Tests*")
     64 
     65            ;; get out of evil insert mode and dont truncate text
     66            (switch-to-buffer "*Ginkgo Tests*")
     67            (local-set-key (kbd "q") 'kill-buffer-and-window)
     68            (evil-motion-state)
     69            (toggle-truncate-lines)
     70            (message nil)
     71            (previous-buffer))
     72 
     73          (defun go-cover ()
     74            (interactive)
     75            (shell-command "go tool cover -html=out.cover"))
