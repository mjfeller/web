     24 (use-package org
     25   :bind (("C-c l" . org-store-link)
     26          ("C-c a" . org-agenda)
     27          ("C-c c" . org-capture)
     28          :map org-src-mode-map
     29          ("C-x C-s" . org-edit-src-exit))
     30   :delight (org-indent-mode nil t)
     31   :config
     32   (progn (require 'org-indent)
     33          (require 'org-src)
     34          (require 'ob-shell)
     35 
     36          (setq org-startup-indented t)
     37          (setq org-src-tab-acts-natively t)
     38          (setq org-src-fontify-natively t)
     39          (setq org-hide-emphasis-markers t)
     40 
     41          (defun org-summary-todo (n-done n-not-done)
     42            "Switch entry to DONE when all subentries are done, to TODO otherwise."
     43            (let (org-log-done org-log-states)   ; turn off logging
     44              (org-todo (if (= n-not-done 0) "DONE" "TODO"))))
     45          (add-hook 'org-after-todo-statistics-hook 'org-summary-todo)
     46 
     47          ;; disable line numbers in org mode
     48          (add-hook 'org-mode-hook (lambda () (org-bullets-mode 1)))
     49 
     50          ;; where I keep all my org files
     51          (setq org-agenda-files
     52                '("~/org/work.org"
     53                  "~/org/life.org"))
     54 
     55          ;; personal capture templates
     56          (setq org-capture-templates
     57                '(
     58                  ;; unfiled work items
     59                  ("w" "Work" entry (file+headline "~/org/work.org" "Unread")
     60                   "* TODO %?\n  %i\n")
     61 
     62                  ;; random thoughts I'd like to capture
     63                  ("t" "Thought" entry (file+headline "~/org/work.org" "Thoughts")
     64                   "* %?\n  %i\n")
     65 
     66                  ;; unfiled work items in my life
     67                  ("l" "Life" entry (file+headline "~/org/life.org" "Inbox")
     68                   "* TODO %?\n  %i\n")
     69 
     70                  ("j" "Journal" entry (file+datetree "~/org/life.org")
     71                   "* TODO %?\n")))
     72 
     73          ;; when refiling an org header don't search more than 3 levels deep
     74          (setq org-refile-targets
     75                '((nil :maxlevel . 3)
     76                  (org-agenda-files :maxlevel . 3)))
     77 
     78          (setq org-todo-keywords
     79                '((sequence "TODO(t)" "BLOCKED(b)" "|" "DONE(d)" "CANCELLED(c)")
     80                  (sequence "RECURRING(r)" "|" "DONE(d)")))
     81 
     82          ;; don't make the title big, I don't like that. Also remove those ugly
     83          ;; checkboxes
     84          (set-face-attribute 'org-document-title nil :height 1)
     85          (set-face-attribute 'org-checkbox nil :box nil)))
     86 
     87 (use-package org-bullets
     88   :config
     89   (progn (setq org-bullets-bullet-list '("●" "○" "◉" "•"))
     90 
     91          ;; bullets on lists
     92          (font-lock-add-keywords
     93           'org-mode
     94           '(("^ +\\([-*]\\) "
     95              (0 (prog1 () (compose-region (match-beginning 1) (match-end 1) "•"))))))))
     96 
     97 (use-package org-fancy-priorities
     98   :ensure t
     99   :delight (org-fancy-priorities-mode)
    100   :hook (org-mode . org-fancy-priorities-mode)
    101   :config
    102   (progn (setq org-fancy-priorities-list '("⬆" "➡" "⬇" "☕"))
    103 
    104           ;; solarized theme on priorities
    105           (setq org-priority-faces '((65 . "#dc322f")
    106                                      (66 . "#cb4b16")
    107                                      (67 . "#859900")))))
