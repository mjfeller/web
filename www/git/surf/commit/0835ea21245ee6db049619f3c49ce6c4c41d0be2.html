+	SoupCookieJar *cookies;
+	SoupCookie *c;
+	SoupDate *e;
+	GSList *l;
+
+	cookies = soup_cookie_jar_text_new(cookiefile, FALSE);
+	for (l = soup_cookies_from_response(m); l; l = l->next){
+		c = (SoupCookie *)l->data;

+	g_slist_free(l);
