     26 #define SETPROP(p) { \
     27         .v = (char *[]){ "/bin/sh", "-c", \
     28                 "prop=\"`xwininfo -children -id $1 | grep '^     0x' |" \
     29                 "sed -e's@^ *\\(0x[0-9a-f]*\\) \"\\([^\"]*\\)\".*@\\1 \\2@' |" \
     30                 "xargs -0 printf %b | dmenu -l 10 -w $1`\" &&" \
     31                 "xprop -id $1 -f $0 8s -set $0 \"$prop\"", \
     32                 p, winid, NULL \
     33         } \
     34 }
