+
+	if(!baseismin) { /* temporarily remove base dimensions */
+		*w -= c->basew;

+	}
+
+	/* adjust for aspect limits */
+	if(c->mina > 0 && c->maxa > 0) {
+		if(c->maxa < (float)*w / *h)

+			*h = *w * c->mina;
+	}
+
+	if(baseismin) { /* increment calculation requires this */
+		*w -= c->basew;
+		*h -= c->baseh;

+	if(c->incw)
+		*w -= *w % c->incw;

+	*w += c->basew;
