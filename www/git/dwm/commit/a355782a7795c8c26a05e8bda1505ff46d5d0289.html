+setmfact(const char *arg) {
+	double delta;
+
+	if(!arg)
+		return;
+	delta = strtod(arg, NULL);
+	if(arg[0] == '-' || arg[0] == '+') {
+		if(mfact + delta < 0.1 || mfact + delta > 0.9)
+			return;
+		mfact += delta;
+	}
+	else {
+		if(delta < 0.1 || delta > 0.9)
+			return;
+		mfact = delta;
+	}
+	arrange();
