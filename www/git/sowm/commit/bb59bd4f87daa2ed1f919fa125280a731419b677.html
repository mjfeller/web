+-+void win_round_corners(Window w, int rad) {
+-+    XWindowAttributes attr2;
+-+    XGetWindowAttributes(d, w, &attr2);
+-+
+-+    int dia = 2 * rad;
+-+    int ww  = attr2.width;
+-+    int wh  = attr2.height;
+-+
+-+    if (ww < dia || wh < dia) return;
+-+
+-+    Pixmap mask = XCreatePixmap(d, w, ww, wh, 1);
+-+
+-+    if (!mask) return;
+-+

+-+    GC shape_gc = XCreateGC(d, mask, 0, &xgcv);
+-+
+-+    if (!shape_gc) {
+-+        XFreePixmap(d, mask);
